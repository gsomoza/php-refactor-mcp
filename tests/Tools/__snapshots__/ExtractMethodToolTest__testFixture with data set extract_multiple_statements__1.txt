<?php

// range: 9-10
// methodName: calculateSum
class Calculator
{
    public function calculate()
    {
        $a = 1;
        $b = 2;
        $sum = $this->calculateSum($a, $b);
        return $sum;
    }
    private function calculateSum($a, $b)
    {
        $c = 3;
        $sum = $a + $b + $c;
        return $sum;
    }
}
